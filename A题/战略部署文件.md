### 4.6 Uncertainty Quantification Method ğŸ”´ MUST (åŸæ–‡: "quantify uncertainty")

```python
def compute_tte_with_uncertainty(model, scenario, n_bootstrap=1000):
    """
    Bootstrap-based TTE uncertainty quantification
    
    Method:
    1. Sample parameter vectors from their distributions
    2. For each sample, solve ODE and compute TTE
    3. Compute percentiles for confidence interval
    """
    tte_samples = []
    
    for i in range(n_bootstrap):
        # Sample parameters with uncertainty
        params_sample = {
            'P_screen': np.random.normal(P_screen_mean, P_screen_std),
            'P_proc': np.random.normal(P_proc_mean, P_proc_std),
            # ... other parameters
        }
        
        # Solve ODE with sampled parameters
        soc_t = solve_ode(model, params_sample, scenario)
        
        # Compute TTE
        tte = find_tte(soc_t, threshold=0.05)
        tte_samples.append(tte)
    
    # Compute confidence interval
    tte_mean = np.mean(tte_samples)
    tte_ci_lower = np.percentile(tte_samples, 2.5)
    tte_ci_upper = np.percentile(tte_samples, 97.5)
    
    return {
        'point_estimate': tte_mean,
        'ci_95': (tte_ci_lower, tte_ci_upper),
        'ci_width': tte_ci_upper - tte_ci_lower
```
